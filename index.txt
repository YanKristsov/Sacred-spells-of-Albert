    <script>
        // --- JavaScript (полностью из вашего файла) ---
        let filterActive = false;
        let allExpanded = false;
        const STORAGE_KEY = 'pathfinder-spells-checked';
        
        function toggleSpell(element) {
            const expandable = element.parentElement.querySelector('.spell-expandable');
            if (expandable.classList.contains('expanded')) {
                expandable.classList.remove('expanded');
            } else {
                expandable.classList.add('expanded');
            }
            if (allExpanded) {
                const visibleExpanded = document.querySelectorAll('.spell-card:not(.hidden) .spell-expandable.expanded').length;
                const visibleTotal = document.querySelectorAll('.spell-card:not(.hidden) .spell-expandable').length;
                if (visibleExpanded === 0) {
                    allExpanded = false;
                    const expandButton = document.getElementById('expandAllButton');
                    expandButton.querySelector('span').textContent = 'Раскрыть все';
                    expandButton.querySelector('i').className = 'fas fa-chevron-down';
                    expandButton.classList.remove('expanded-all');
                }
            }
        }
        
        function toggleCheck(button) {
            button.classList.toggle('checked');
            saveAllCheckedSpells();
            if (filterActive) {
                hideUncheckedSpells();
            }
            resetExpandButton();
        }
        
        function saveAllCheckedSpells() {
            const checkedSpells = [];
            document.querySelectorAll('.spell-card').forEach(card => {
                const checkButton = card.querySelector('.check-button');
                const spellName = card.querySelector('.spell-title').textContent.trim();
                if (checkButton && checkButton.classList.contains('checked')) {
                    checkedSpells.push(spellName);
                }
            });
            localStorage.setItem(STORAGE_KEY, JSON.stringify(checkedSpells));
        }
        
        function loadCheckedSpells() {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (!saved) return;
            try {
                const checkedSpells = JSON.parse(saved);
                document.querySelectorAll('.spell-card').forEach(card => {
                    const checkButton = card.querySelector('.check-button');
                    const spellName = card.querySelector('.spell-title').textContent.trim();
                    if (checkedSpells.includes(spellName)) {
                        checkButton.classList.add('checked');
                    }
                });
            } catch (e) {
                console.error('Ошибка загрузки:', e);
            }
        }
        
        function toggleFilter() {
            filterActive = !filterActive;
            const filterButton = document.getElementById('filterButton');
            const filterText = filterButton.querySelector('span');
            const filterIcon = filterButton.querySelector('i');
            
            if (filterActive) {
                filterText.textContent = 'Все заклинания';
                filterIcon.className = 'fas fa-eye';
                filterButton.classList.add('filter-active');
                hideUncheckedSpells();
            } else {
                filterText.textContent = 'Отмеченные';
                filterIcon.className = 'fas fa-filter';
                filterButton.classList.remove('filter-active');
                showAllSpells();
            }
            localStorage.setItem('filter-active', filterActive);
            resetExpandButton();
        }
        
        function hideUncheckedSpells() {
            document.querySelectorAll('.spell-card').forEach(card => {
                const checkButton = card.querySelector('.check-button');
                if (!checkButton.classList.contains('checked')) {
                    card.classList.add('hidden');
                } else {
                    card.classList.remove('hidden');
                }
            });
        }
        
        function showAllSpells() {
            document.querySelectorAll('.spell-card').forEach(card => {
                card.classList.remove('hidden');
            });
        }
        
        function loadFilterState() {
            const savedFilter = localStorage.getItem('filter-active');
            if (savedFilter === 'true') {
                filterActive = true;
                const filterButton = document.getElementById('filterButton');
                filterButton.querySelector('span').textContent = 'Все заклинания';
                filterButton.querySelector('i').className = 'fas fa-eye';
                filterButton.classList.add('filter-active');
                hideUncheckedSpells();
            }
        }
        
        function resetExpandButton() {
            allExpanded = false;
            const expandButton = document.getElementById('expandAllButton');
            expandButton.querySelector('span').textContent = 'Раскрыть все';
            expandButton.querySelector('i').className = 'fas fa-chevron-down';
            expandButton.classList.remove('expanded-all');
        }
        
        function expandVisibleSpells() {
            document.querySelectorAll('.spell-card:not(.hidden) .spell-expandable').forEach(expandable => {
                expandable.classList.add('expanded');
            });
        }
        
        function collapseVisibleSpells() {
            document.querySelectorAll('.spell-card:not(.hidden) .spell-expandable').forEach(expandable => {
                expandable.classList.remove('expanded');
            });
        }
        
        function toggleExpandAll() {
            const expandButton = document.getElementById('expandAllButton');
            const expandText = expandButton.querySelector('span');
            const expandIcon = expandButton.querySelector('i');
            
            if (allExpanded) {
                collapseVisibleSpells();
                expandText.textContent = 'Раскрыть все';
                expandIcon.className = 'fas fa-chevron-down';
                expandButton.classList.remove('expanded-all');
            } else {
                expandVisibleSpells();
                expandText.textContent = 'Свернуть все';
                expandIcon.className = 'fas fa-chevron-up';
                expandButton.classList.add('expanded-all');
            }
            allExpanded = !allExpanded;
        }
        
        document.addEventListener('DOMContentLoaded', function() {
            loadCheckedSpells();
            loadFilterState();
            const filterBtn = document.getElementById('filterButton');
            if (filterBtn) filterBtn.addEventListener('click', toggleFilter);
            const expandBtn = document.getElementById('expandAllButton');
            if (expandBtn) expandBtn.addEventListener('click', toggleExpandAll);
        });
        
        document.addEventListener('click', function(event) {
            if (!event.target.closest('.spell-card') && !event.target.closest('.filter-button') && !event.target.closest('.expand-all-button')) {
                document.querySelectorAll('.spell-expandable.expanded').forEach(item => {
                    item.classList.remove('expanded');
                });
                resetExpandButton();
            }
        });
    </script>
